# SGIE secondary classifier — color of jockey torso (5 classes).
#
# Input: 3x128x128 (color_classifier_v4 architecture)
# Output: 5 logits → blue, green, purple, red, yellow
#
# Net is trained with ImageNet normalization:
#   x = (pixel/255 - mean) / std,  mean=[0.485,0.456,0.406], std=[0.229,0.224,0.225]
# nvinfer only supports x = (pixel - offsets) * net-scale-factor,
# so we approximate with channel-averaged std (0.226):
#   net-scale-factor = 1/(255*0.226) = 0.01735
#   offsets          = mean*255       = [123.675, 116.28, 103.53]   (RGB)
# This is the same approximation used by all sample resnet sgies.

[property]
gpu-id=0

# Model
onnx-file=/home/ipodrom/Рабочий стол/Ipodrom-Project/user/race_vision/models/color_classifier_v4.onnx
model-engine-file=/home/ipodrom/Рабочий стол/Ipodrom-Project/user/race_vision/models/color_classifier_v4.engine
labelfile-path=/home/ipodrom/Рабочий стол/Ipodrom-Project/user/race_vision/deepstream/configs/labels_color.txt

# 0=FP32, 1=INT8, 2=FP16
network-mode=2
batch-size=64

# Preprocessing — approximate ImageNet normalization (RGB)
net-scale-factor=0.01735
offsets=123.675;116.28;103.53
model-color-format=0

# Network type: 1 = classifier
network-type=1
process-mode=2

# Operate on any upstream GIE (−1), class 0 (person). -1 avoids silent miss
# if PGIE gie-unique-id isn't explicitly propagated.
gie-unique-id=2
operate-on-gie-id=-1
operate-on-class-ids=0

# Filters — jockey bbox at 1280x720 mux is ~25x60
# Lower bound = reject YOLO noise; upper bound = reject spectators/staff
input-object-min-width=12
input-object-min-height=40
input-object-max-width=80
input-object-max-height=180

classifier-async-mode=0
classifier-threshold=0.5

# Tensor I/O
infer-dims=3;128;128
output-blob-names=output

# Custom classifier parser (built-in auto-parse does not work with our 2D output)
parse-classifier-func-name=NvDsInferClassiferParseCustomColor
custom-lib-path=/home/ipodrom/Рабочий стол/Ipodrom-Project/user/race_vision/deepstream/build/libnvdsinfer_racevision.so
